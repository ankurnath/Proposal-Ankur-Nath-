\begin{frame}{The Scalability Challenge}
  \begin{itemize}
    \item Modern real-world graphs (social networks, web graphs, biological networks)
      contain \textbf{millions or billions} of nodes
    \item The optimal solution often involves only a \textbf{tiny fraction} of the
      ground set (e.g., selecting a small seed set from an entire social network)
    \item Yet existing solvers must \textbf{evaluate every candidate}, leading to
      prohibitive runtimes
    \item Even the best polynomial-time heuristics become
      \textbf{prohibitively slow} as graphs grow
  \end{itemize}

  \vspace{0.4cm}

  \begin{alertblock}{The Mismatch}
    Enormous ground set $|\mathcal{U}|$ vs.\ tiny optimal solution $|X^\star|$
    $\Rightarrow$ fundamental scalability bottleneck. Most elements are irrelevant,
    yet we pay the cost of considering all of them.
  \end{alertblock}
\end{frame}

\begin{frame}{Search Space Pruning}
  \textbf{Goal:} Before running the expensive solver, remove elements that are unlikely to
  be part of an optimal solution.

  \vspace{0.3cm}

  \begin{block}{Formal Problem}
    Given ground set $\mathcal{U}$, objective $f$, and constraints $\Omega$, construct
    $\mathcal{U}' \subseteq \mathcal{U}$ with $|\mathcal{U}'| \ll |\mathcal{U}|$ such that:
    \[
      \frac{f(\mathcal{H}(\mathcal{U}'))}{f(\mathcal{H}(\mathcal{U}))} \approx 1
    \]
    where $\mathcal{H}$ is any downstream heuristic solver.
  \end{block}

  \vspace{0.3cm}

  \textbf{Evaluation Metrics:}
  \begin{itemize}
    \item $P_r = f(\mathcal{H}(\mathcal{U}')) / f(\mathcal{H}(\mathcal{U}))$
      \hfill \textit{solution quality preserved}
    \item $P_g$ = fraction of ground set removed
      \hfill \textit{how much we pruned}
    \item $C = P_r \cdot P_g$
      \hfill \textit{combined metric (balances both)}
  \end{itemize}
\end{frame}

\begin{frame}{Limitations of Existing Approaches}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{Classical Methods}
        \begin{itemize}
          \item Submodular Sparsification~\cite{zhou2017scaling}: sequential, $O(n^2)$ pairwise computations
          \item \textbf{No theoretical guarantees} on pruning quality before our work
          \item Do not scale to million-node graphs
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Learning-Based Methods}
        \begin{itemize}
          \item GCOMB-P~\cite{manchanda2020gcomb}: handcrafted degree heuristic
          \item LeNSE~\cite{ireland2022lense}: requires domain expertise for subgraph categorization
          \item COMBHelper~\cite{tian2024combhelper}: expensive Fourier Feature Mapping
          \item \textbf{All designed for size constraints only}
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.4cm}

  \centering
  \textbf{Common bottleneck:} reliance on human-designed features and problem-specific
  algorithmic choices.
\end{frame}

\begin{frame}{Thesis Contributions}
  \begin{enumerate}
    \item \textbf{QuickPrune}~\cite{pmlr-v258-nath25a} \textit{(AISTATS 2025, published)}\\
      First pruning algorithm with \textbf{provable theoretical guarantees} on both
      solution quality and pruned set size. Single-pass, works under size and knapsack
      constraints.

      \vspace{0.2cm}

    \item \textbf{Hierarchical DeepPruner}~\cite{nath2025hdeeppruner} \textit{(SoCS 2025, published)}\\
      Adaptive two-stage framework: \textbf{GNN pre-pruning} + \textbf{Neural MCTS
      fine-pruning}. Eliminates handcrafted features entirely; captures element
      interdependencies.

      \vspace{0.2cm}

    \item \textbf{LLM2Prune} \textit{(Proposed work)}\\
      LLM-directed beam search for \textbf{automated feature discovery}.
      Constraint-adaptive (size $\to$ knapsack by changing a text prompt). Orders of
      magnitude faster at inference.
  \end{enumerate}

  \vspace{0.2cm}

  \begin{alertblock}{Common Result}
    All three methods prune graphs with up to 1.7 million nodes, removing over 90\% of the
    ground set while preserving near-optimal solution quality.
  \end{alertblock}
\end{frame}

\section{QuickPrune (AISTATS 2025)}

\begin{frame}{QuickPrune: The Problem}
  \textbf{Before our work:} all existing pruning methods were heuristics with
  \textbf{no formal performance guarantees}.

  \vspace{0.3cm}

  \textbf{Open question:} Is it even possible to efficiently produce a pruned ground set
  with provable bounds on both its size and the fraction of optimal value it retains?

  \vspace{0.3cm}

  \begin{block}{Our Answer: Yes}
    QuickPrune is a \textbf{lightweight, single-pass} algorithm that:
    \begin{itemize}
      \item Processes each element \textbf{exactly once} (streaming)
      \item Provides \textbf{theoretical guarantees} on $P_r$ and $|\mathcal{U}'|$
      \item Works under \textbf{$\gamma$-weak submodularity} (generalizes submodularity)
      \item Handles both \textbf{size and knapsack} constraints
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{QuickPrune: Key Assumptions}
  \begin{block}{$\gamma$-Weak Submodularity}
    A set function $f$ is $\gamma$-weakly submodular if for all $S \subseteq T$:
    \[
      f(e \mid S) \geq \gamma \cdot f(e \mid T), \quad \gamma \in (0, 1]
    \]
    When $\gamma = 1$, this is standard submodularity (diminishing returns).
    Many real-world objectives (IM, MaxCover) are fully submodular; MaxCut is
    $\gamma$-weakly submodular.
  \end{block}

  \vspace{0.2cm}

  \begin{block}{No-Heavy-Items (NHI) Assumption}
    No single element has a marginal contribution exceeding an $\eta$-fraction of the
    optimal value. This is mild: in large graphs, individual nodes rarely dominate.
  \end{block}
\end{frame}

\begin{frame}{QuickPrune-Single: How It Works}
  For a single budget $\kappa$, stream through $\mathcal{U}$ one element at a time:

  \vspace{0.2cm}

  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{enumerate}
        \item \textbf{Skip} if $\text{cost}(e) > \kappa$
        \item \textbf{Addition rule:} Keep $e$ if its marginal gain is ``cost-effective'':
          \[
            \Delta(e \mid A) \geq \frac{\delta \cdot c(e) \cdot f(A)}{\kappa}
          \]
        \item \textbf{Deletion step:} When $f(A)$ grows by factor $n/\epsilon$ since last
          checkpoint, discard old elements
        \item Track best singleton $a^*$
        \item Return $\mathcal{U}' = A \cup \{a^*\}$
      \end{enumerate}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{exampleblock}{Intuition}
        \textbf{Addition:} Only keep elements whose contribution justifies their cost.

        \vspace{0.2cm}

        \textbf{Deletion:} As better elements arrive, earlier ``good enough'' elements
        become relatively weak. Periodic cleanup keeps $|\mathcal{U}'|$ small.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{QuickPrune: Multi-Budget Extension}
  \textbf{Problem:} QuickPrune-Single works for one budget $\kappa$. In practice, we
  want a \textbf{single} pruned set that works for \textbf{any} budget in a range
  $[\kappa_{\min}, \kappa_{\max}]$.

  \vspace{0.3cm}

  \textbf{Solution:} Run copies of QuickPrune-Single in parallel for a geometric set of
  budgets:
  \[
    \mathcal{B} = \left\{\kappa_{\max}(1-\eta)^i : (1-\eta)\kappa_{\min} \leq
    \kappa_{\max}(1-\eta)^i \leq \kappa_{\max}\right\}
  \]
  Return the \textbf{union} of all pruned sets.

  \vspace{0.3cm}

  \begin{block}{Complexity}
    \begin{itemize}
      \item $\mathcal{O}(\log(\kappa_{\max}/\kappa_{\min}))$ copies running in parallel
      \item Same number of oracle queries \textbf{per element} --- still a single pass
        through $\mathcal{U}$
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{QuickPrune: Theoretical Guarantees}
  \begin{block}{Theorem 1 (Single Budget)}
    Under $\gamma$-weak submodularity, QuickPrune-Single produces $\mathcal{U}'$ with:
    \begin{itemize}
      \item \textbf{Size:} $|\mathcal{U}'| < 2\!\left(1 + \frac{\kappa}{\delta \,
        c_{\min}}\right)\!\log(n/\epsilon) + 3$
      \item \textbf{Value:} $\exists\, A' \subseteq \mathcal{U}'$: \;
        $f(A') \geq \frac{\delta \gamma^4(1 - \epsilon\gamma^{-1})}
        {2(\delta\gamma^2 + 1)(1 + \gamma^{-1}\delta)} \cdot f_\kappa(\mathcal{U})$
    \end{itemize}
  \end{block}

  \begin{block}{Theorem 2 (Multi-Budget --- Main Result)}
    For all $\kappa' \in [\kappa_{\min}, \kappa_{\max}]$ simultaneously:
    \begin{itemize}
      \item \textbf{Value:} $f_{\kappa'}(\mathcal{U}') \geq \Omega(\gamma^5) \cdot
        f_{\kappa'}(\mathcal{U})$
      \item \textbf{Size:} $|\mathcal{U}'| = \mathcal{O}\!\left(
        \log\!\left(\frac{\kappa_{\max}}{\kappa_{\min}}\right)
        \!\left(1 + \frac{\kappa_{\max}}{\delta \, c_{\min}}\right)
        \!\log(n / \epsilon)\right)$
    \end{itemize}
  \end{block}

  \vspace{0.1cm}
  \centering
  Pruned set size is \textbf{logarithmic} in the original ground set size $n$.
\end{frame}

\section{H-DeepPruner (SoCS 2025)}

\begin{frame}{H-DeepPruner: The Problem}
  \textbf{Can we design a pruning method that learns without expert algorithm design?}

  \vspace{0.3cm}

  \textbf{Limitations of existing learning-based methods:}
  \begin{itemize}
    \item GCOMB-P~\cite{manchanda2020gcomb}, LeNSE~\cite{ireland2022lense}, COMBHelper~\cite{tian2024combhelper} all require \textbf{problem-specific domain
      knowledge} or handcrafted features
    \item Difficult to apply to a new CO problem without significant redesign
  \end{itemize}

  \vspace{0.3cm}

  \begin{block}{Our Approach}
    H-DeepPruner: an adaptive two-stage framework combining \textbf{supervised learning}
    (fast, coarse pruning) with \textbf{reinforcement learning} (fine-grained refinement
    capturing element interactions). No handcrafted features needed.
  \end{block}
\end{frame}

\begin{frame}{H-DeepPruner: Framework Overview}
  \begin{figure}
    \centering
    \includegraphics[width=0.85\textwidth]{figures/hdeeppruner/framework.pdf}
    \caption{H-DeepPruner: Stage 1 (GNN) quickly eliminates clearly irrelevant nodes.
      Stage 2 (MCTS) refines the pruned set by considering element interdependencies.}
  \end{figure}
\end{frame}

\begin{frame}{H-DeepPruner: Stage 1 --- GNN Pre-Pruning}
  \textbf{Goal:} Quickly eliminate the bulk of irrelevant nodes.

  \vspace{0.3cm}

  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item Train a Graph Convolutional Network (GCN) for \textbf{binary
          classification}: predict whether each node belongs in the optimal solution
        \item \textbf{Input features:} random node initialization --- random features
          break symmetry and allow the GNN to distinguish structurally equivalent nodes
        \item \textbf{No handcrafted features needed} --- the GNN learns to extract
          relevant structural information directly from the graph
        \item \textbf{Training labels:} obtained by running a heuristic solver on small
          training graphs
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{exampleblock}{Why Random Features?}
        Standard GNNs cannot distinguish nodes with identical neighborhoods. Random
        initialization provides unique identifiers, enabling the GNN to learn
        position-aware representations without requiring domain-specific feature
        engineering.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{H-DeepPruner: Stage 2 --- Neural MCTS Fine-Pruning}
  \textbf{Problem:} The GNN treats nodes independently, missing element interactions.

  \vspace{0.2cm}

  \textbf{Solution:} Formulate pruning as a \textbf{Markov Decision Process} and solve
  with Monte Carlo Tree Search.

  \vspace{0.2cm}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{MDP Formulation:}
      \begin{itemize}
        \item \textbf{State:} current subset of retained nodes
        \item \textbf{Action:} add or remove a node
        \item \textbf{Reward:} combined metric $C$ at terminal state
        \item \textbf{Transition:} deterministic state update
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{MCTS Phases:}
      \begin{itemize}
        \item \textbf{Selection:} PUCT balances exploration vs.\ exploitation
        \item \textbf{Expansion:} progressive widening adds children gradually
        \item \textbf{Simulation:} evaluate via heuristic
        \item \textbf{Backpropagation:} update value estimates
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3cm}

  \textbf{Progressive deepening} iteratively increases search depth, allowing the
  algorithm to refine its decisions as it explores deeper into the search tree.
\end{frame}

\begin{frame}{H-DeepPruner: MCTS Illustration}
  \begin{figure}
    \centering

    \caption{Neural MCTS with progressive widening: the tree expands gradually, balancing

  \end{figure}
\end{frame}

\section{LLM2Prune (Proposed Work)}

\begin{frame}{LLM2Prune: The Problem}
  \textbf{Despite progress, a key bottleneck remains:}

  \vspace{0.2cm}

  \begin{itemize}
    \item \textbf{Classical methods} (QuickPrune, SS): sequential processing limits
      scalability on very large graphs
    \item \textbf{Learning-based methods} (H-DeepPruner, GCOMB-P, LeNSE, COMBHelper):
      still require some form of \textbf{handcrafted design choices} (features, subgraph
      sampling, boosting modules)
    \item \textbf{Constraint limitation:} most learning-based methods only work under size
      constraints; adapting to knapsack requires significant redesign
  \end{itemize}

  \vspace{0.3cm}

  \begin{block}{Key Question}
    Can we design a pruning framework that \textbf{avoids all handcrafted features},
    delivers strong performance across \textbf{general CO problems over graphs}, and
    adapts to \textbf{different constraints} without any algorithmic changes?
  \end{block}
\end{frame}

\begin{frame}{LLM2Prune: Main Idea}
  \textbf{Insight:} Large language models encode broad domain knowledge about graph theory,
  optimization, and network science. We can leverage this knowledge to \textbf{automate
  feature discovery}.

  \vspace{0.3cm}

  \textbf{How it works (in simple terms):}
  \begin{enumerate}
    \item Give the LLM a \textbf{text description} of the CO problem and constraints
    \item The LLM \textbf{proposes node-level features} (e.g., ``degree-to-weight ratio'')
    \item The LLM \textbf{writes Python code} to compute each feature
    \item A lightweight GCN is trained on these features to classify nodes
    \item \textbf{Feedback} (which features helped, which didn't) is sent back to the LLM
    \item The process repeats via \textbf{beam search} to find the best feature set
  \end{enumerate}

  \vspace{0.2cm}

  \begin{exampleblock}{Key Advantage}
    To adapt from size to knapsack constraints: just \textbf{change the task description
    text}. No code changes, no retraining of the search process.
  \end{exampleblock}
\end{frame}

\begin{frame}{LLM2Prune: Framework}
  \begin{figure}
    \centering
    \includegraphics[width=0.92\textwidth]{figures/llm2prune/LLM_cropped.pdf}
    \caption{(a) Beam search with expansion and pruning phases. (b) Expansion: LLM
      proposes features $\to$ generates code $\to$ trains GCN $\to$ evaluates performance
      and feature importance $\to$ feeds back to LLM.}
  \end{figure}
\end{frame}

\begin{frame}{LLM2Prune: Decision Tree Formulation}
  The feature search is formalized as a \textbf{decision tree} $\mathcal{T}$:

  \vspace{0.2cm}

  Each node $n_{d,i}$ at depth $d$ stores:
  \[
    n_{d,i} = \big(\underbrace{\mathcal{F}_{d,i}}_{\text{features}},\;
    \underbrace{\mathcal{M}_{d,i}}_{\text{performance}},\;
    \underbrace{g_{\theta_{d,i}}}_{\text{classifier}},\;
    \underbrace{\mathcal{H}_{d,i}}_{\text{history}}\big)
  \]

  \vspace{0.2cm}

  \textbf{Cumulative history} $\mathcal{H}_{d,i}$ records all past attempts along the
  root-to-node path:
  \[
    \mathcal{H}_{d,i} = \bigcup_{\ell=0}^{d} \bigcup_{j \in \pi(\ell,i)}
    \Big(\mathcal{F}_{\ell,j},\; M_{\ell,j},\; \text{Imp}_{\ell,j},\;
    \mathcal{F}^{\text{fail}}_{\ell,j}\Big)
  \]

  \vspace{0.2cm}

  \begin{block}{Why Full Path History?}
    Conditioning on the full root-to-node path (not just the parent) prevents the LLM
    from re-proposing features that were explored and discarded at earlier depths.
  \end{block}
\end{frame}

\begin{frame}{LLM2Prune: Expansion and Pruning}
  \textbf{At each depth $d$, for each node in beam $B_d$:}

  \vspace{0.2cm}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{Expansion Phase}
        \begin{enumerate}
          \item LLM \textbf{summarizes} history into key insights
          \item LLM \textbf{proposes} $\gamma$ child feature sets guided by refinement
            rules
          \item LLM \textbf{generates code} for each feature (timeout $\tau$ discards
            expensive ones)
          \item Train GCN with \textbf{semi-supervised learning} (soft pseudo-labels
            mitigate label noise)
          \item \textbf{GNNExplainer} extracts feature importance scores
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Pruning Phase}
        \begin{itemize}
          \item Evaluate all candidate children by combined metric $M$
          \item Keep top-$\beta$ nodes $\to$ next beam $B_{d+1}$
          \item Continue until max depth $D_{\max}$
          \item Return globally best node:
            \[
              n^{\text{best}} = \arg\max_{n \in \mathcal{T}} M
            \]
          \item Fine-tune best classifier on target graph
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{LLM2Prune: Preliminary Results (Size Constraint)}
  \begin{itemize}
    \item Competitive with classical methods (QuickPrune, SS) on combined metric
    \item \textbf{1--4 orders of magnitude faster} at inference on large graphs
    \item Consistently \textbf{outperforms all learning-based baselines}
  \end{itemize}

  \begin{figure}
    \centering
    \includegraphics[width=0.65\textwidth]{figures/llm2prune/runtimes_radar.pdf}
    \caption{Inference runtime (log scale) across eight datasets and three applications.

  \end{figure}
\end{frame}

\begin{frame}{LLM2Prune: Discovered Features}
  LLM2Prune discovers features that are \textbf{grounded in the problem objective}, not
  generic graph statistics:

  \vspace{0.3cm}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{Maximum Cut}
        \textbf{Random cut expectation:} expected marginal contribution of a node to the
        cut under a random partition. Directly derived from the cut objective.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Influence Maximization}
        \textbf{Avg.\ incoming activation probability:} mean probability a node is
        activated by neighbors (susceptibility).\\
        \textbf{Outgoing activation probability sum:} captures direct influence spread.
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.3cm}

  \centering
  These features would require significant domain expertise to design manually,\\
  yet LLM2Prune discovers them \textbf{automatically from the task description alone}.
\end{frame}

\begin{frame}{LLM2Prune: Ablation Study}
  \begin{columns}
    \begin{column}{0.42\textwidth}
      Compared four variants with the \textbf{same number of LLM API calls}:
      \begin{enumerate}
        \item Single-shot prompt
        \item Feedback loop (performance)
        \item Feedback loop (performance + importance)
        \item \textbf{Beam search}
      \end{enumerate}

      \vspace{0.2cm}

      Beam search achieves \textbf{higher and more stable} performance. Simpler variants
      suffer from \textbf{error propagation} when early iterations contain reasoning
      errors.
    \end{column}
    \begin{column}{0.55\textwidth}
      \begin{figure}
        \centering
        \includegraphics[width=\textwidth]{figures/llm2prune/ablation.pdf}
        \caption{Combined metric across three CO problems. Beam search has median close
          to 1.0 with minimal variance.}
      \end{figure}
    \end{column}
  \end{columns}
\end{frame}

\section{Summary}

\begin{frame}{Comparison of All Three Methods}
  \begin{table}
    \centering
    \small
    \begin{tabular}{lccc}
      \toprule
      & \textbf{QuickPrune} & \textbf{H-DeepPruner} & \textbf{LLM2Prune} \\
      \midrule
      Approach & Classical & Supervised + RL & LLM + Supervised \\
      Handcrafted features & No & No & No \\
      Theoretical guarantees & \textbf{Yes} & No & No \\
      Size constraint & Yes & Yes & Yes \\
      Knapsack constraint & Yes & No & Yes \\
      Inference speed & Moderate & Fast & \textbf{Very fast} \\
      Element interactions & No & \textbf{Yes} (MCTS) & No \\
      Feature discovery & N/A & Random init & \textbf{Automated} (LLM) \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.3cm}

  The three methods are \textbf{complementary}: QuickPrune provides theoretical
  foundations, H-DeepPruner captures element interactions, and LLM2Prune automates feature
  discovery with constraint adaptability.
\end{frame}
